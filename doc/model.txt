Lobby: {
  state: GameState,
  max_lives: int,
  host: int, // set to host id
  board_size: [int, int],
  timer_setting: int,
  lobby_code: str,
  players: Player[],
}

GameState: {
  turns: Turns,
  board: Board,
  timer: int,
  memory: str[], // assumed to be a set
  curr_turn: str // player_id
}

Board: {
  tiles: str[][]
}

Player: {
  id: string, // connection-id
  name: string,
  is_spectator: boolean,
  lives: int,
  money: int
}

Bot(Player): {
  difficulty: int,
  memory: str[], // assumed to be a set
}

Action: {
  action: str,
  player_id: str,
  data: any
}

Action: {
  timestamp: number,
  action: str,
  player_id: str,
  data: any
}
